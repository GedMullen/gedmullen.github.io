package uk.ac.forthvalley.staff;

/**
 * This class represents lecturer instances.
 * 
 * @author Ged Mullen
 * @version 1.0
 */
public class Lecturer extends Person implements Qualifications, Experience {

	public static void main(String[] args) {
		Lecturer gedMullen = new Lecturer("Ged Mullen", Gender.MALE);
		gedMullen.setProfilePictureDescription("TAPS - Timetabling Application Service");
		gedMullen.setSubject("Software Development");
		gedMullen.setCollege("Forth Valley College, Scotland");
		String[] qualifications = { "BSc Information Systems, Staffordshire University (1990)",
				"MSc Ethical Hacking, Abertay University (2011)" };
		gedMullen.setQualifications(qualifications);
		String[] certifications = { "Java Programmer (1996)", "Java Architect (2003)",
				"Java Web Component Developer (2006)" };
		gedMullen.setCertifications(certifications);
		String[] companiesWorkedFor = { "Digital", "British Steel", "Standard Life", "IBM", 
				"Thames Water", "Smile.co.uk", "IntelligentFinance.com", "RBS", "JTechnologies" };
		gedMullen.setCompanies(companiesWorkedFor);
		String[] interests = { "Football", "More Football", "Middlesbrough FC" };
		gedMullen.setInterests(interests);
	}

	private String subject;
	private String college;
	private String[] qualifications;
	private String[] certifications;
	private String[] companies;

	public Lecturer(String name, Gender gender) {
		super(name, gender);
	}

	public String getSubject() {
		return subject;
	}

	public void setSubject(String subject) {
		this.subject = subject;
	}

	public String getCollege() {
		return college;
	}

	public void setCollege(String college) {
		this.college = college;
	}

	public String[] getQualifications() {
		return qualifications;
	}

	public void setQualifications(String[] qualifications) {
		this.qualifications = qualifications;
	}

	public String[] getCertifications() {
		return certifications;
	}

	public void setCertifications(String[] certifications) {
		this.certifications = certifications;
	}

	public String[] getCompanies() {
		return companies;
	}

	public void setCompanies(String[] companies) {
		this.companies = companies;
	}

}

class Person {

	private String name;
	private Gender gender;
	private String profilePictureDescription;
	private String[] interests;

	public Person(String name, Gender gender) {
		this.setName(name);
		this.setGender(gender);
	}

	public String getName() {
		return name;
	}

	private void setName(String name) {
		this.name = name;
	}

	public Gender getGender() {
		return gender;
	}

	private void setGender(Gender gender) {
		this.gender = gender;
	}

	public String getProfilePictureDescription() {
		return profilePictureDescription;
	}

	public void setProfilePictureDescription(String profilePictureDescription) {
		this.profilePictureDescription = profilePictureDescription;
	}

	public String[] getInterests() {
		return interests;
	}

	public void setInterests(String[] interests) {
		this.interests = interests;
	}

}

interface Qualifications {

	public String[] getQualifications();

	public void setQualifications(String[] qualifications);

	public String[] getCertifications();

	public void setCertifications(String[] certifications);

}

interface Experience {

	public String[] getCompanies();

	public void setCompanies(String[] companies);

}

enum Gender {
	MALE, FEMALE
}
